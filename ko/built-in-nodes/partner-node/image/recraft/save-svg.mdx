![ComfyUI 내장 SVG 저장 노드](/images/built-in-nodes/api_nodes/recraft/save-svg.jpg)

SVG 저장 노드는 Recraft 벡터 생성 노드에서 출력된 SVG 데이터를 파일 시스템의 사용 가능한 파일로 저장할 수 있게 해줍니다. 이는 벡터 그래픽을 처리하고 내보내는 데 필수적인 구성 요소입니다.

## 노드 기능

이 노드는 SVG 벡터 데이터를 받아 파일 시스템에 표준 SVG 파일로 저장합니다. 자동 파일 이름 지정과 저장 경로 지정을 지원하며, 이를 통해 다른 소프트웨어에서 벡터 그래픽을 열고 편집할 수 있습니다.

## 매개변수

### 기본 매개변수

| 매개변수       | 유형    | 기본값   | 설명                                                                  |
| --------------- | ------- | --------- | ----------------------------------------------------------------------- |
| svg             | SVG     | -         | 저장할 SVG 벡터 데이터                                                  |
| filename_prefix | 문자열  | "recraft" | 파일 이름의 접두사                                                     |
| output_dir      | 문자열  | -         | 출력 디렉토리, 기본값은 ComfyUI의 출력 폴더인 `ComfyUI/output/svg/` 입니다 |
| index           | 정수    | -1        | 저장 인덱스, -1은 생성된 모든 SVG를 저장함을 의미합니다               |

### 출력

| 출력 | 유형 | 설명                       |
| ------ | ---- | -------------------------- |
| SVG    | SVG  | 입력된 SVG 데이터를 그대로 전달 |

class SaveSVGNode:
    """
    디스크에 SVG 파일을 저장합니다.
    """

    def __init__(self):
        self.output_dir = folder_paths.get_output_directory()
        self.type = "output"
        self.prefix_append = ""

    RETURN_TYPES = ()
    DESCRIPTION = cleandoc(__doc__ or "")  # None 값 처리
    FUNCTION = "save_svg"
    CATEGORY = "api node/image/Recraft"
    OUTPUT_NODE = True

    @classmethod
    def INPUT_TYPES(s):
        return {
            "required": {
                "svg": (RecraftIO.SVG,),
                "filename_prefix": ("STRING", {"default": "svg/ComfyUI", "tooltip": "저장할 파일의 접두사입니다. %date:yyyy-MM-dd% 또는 %Empty Latent Image.width%와 같은 노드 값도 포함할 수 있습니다."})
            },
            "hidden": {
                "prompt": "PROMPT",
                "extra_pnginfo": "EXTRA_PNGINFO"
            }
        }

    def save_svg(self, svg: SVG, filename_prefix="svg/ComfyUI", prompt=None, extra_pnginfo=None):
        filename_prefix += self.prefix_append
        full_output_folder, filename, counter, subfolder, filename_prefix = folder_paths.get_save_image_path(filename_prefix, self.output_dir)
        results = list()

        # 메타데이터 JSON 준비
        metadata_dict = {}
        if prompt is not None:
            metadata_dict["prompt"] = prompt
        if extra_pnginfo is not None:
            metadata_dict.update(extra_pnginfo)

        # 메타데이터를 JSON 문자열로 변환
        metadata_json = json.dumps(metadata_dict, indent=2) if metadata_dict else None

        for batch_number, svg_bytes in enumerate(svg.data):
            filename_with_batch_num = filename.replace("%batch_num%", str(batch_number))
            file = f"{filename_with_batch_num}_{counter:05}_.svg"

            # SVG 콘텐츠 읽기
            svg_bytes.seek(0)
            svg_content = svg_bytes.read().decode('utf-8')

            # 메타데이터가 있으면 삽입
            if metadata_json:
                # CDATA 섹션을 포함한 메타데이터 요소 생성
                metadata_element = f"""  <metadata>
    <![CDATA[
{metadata_json}
    ]]>
  </metadata>
"""
                # 정규표현식을 이용해 svg 태그 이후에 메타데이터 삽입
                import re
                svg_content = re.sub(r'(<svg[^>]*>)', r'\1\n' + metadata_element, svg_content)

            # 수정된 SVG를 파일에 쓰기
            with open(os.path.join(full_output_folder, file), 'wb') as svg_file:
                svg_file.write(svg_content.encode('utf-8'))

            results.append({
                "filename": file,
                "subfolder": subfolder,
                "type": self.type
            })
            counter += 1
        return { "ui": { "images": results } }
