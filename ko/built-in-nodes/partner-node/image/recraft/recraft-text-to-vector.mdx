![ComfyUI 기본 Recraft 텍스트를 벡터로 재구성 노드](/images/built-in-nodes/api_nodes/recraft/recraft-text-to-vector.jpg)

Recraft 텍스트를 벡터로 재구성 노드는 Recraft의 API를 사용해 텍스트 설명을 기반으로 고품질 벡터 그래픽(SVG 형식)을 만들 수 있습니다. 로고, 아이콘 및 무한히 확장 가능한 일러스트레이션을 만드는 데 완벽합니다.

## 매개변수

### 기본 매개변수

| 매개변수 | 유형    | 기본값   | 설명                                        |
| --------- | ------- | --------- | -------------------------------------------------- |
| 프롬프트    | 문자열  | ""        | 생성할 벡터 그래픽의 텍스트 설명                 |
| 서브스타일  | 선택    | -         | 벡터 스타일 하위 유형                               |
| 크기      | 선택    | 1024x1024 | 벡터 출력용 캔버스 크기                  |
| n         | 정수    | 1         | 생성할 결과 수 (1~6)                |
| 시드      | 정수    | 0         | 난수 시드 값                                  |

### 옵션 매개변수

| 매개변수        | 유형             | 설명                                  |
| ---------------- | ---------------- | -------------------------------------------- |
| 네거티브 프롬프트  | 문자열           | 생성에서 제외할 요소들          |
| recraft_controls | Recraft 컨트롤    | 추가 제어 매개변수 (색상 등) |

### 출력

| 출력 | 유형   | 설명                                                   |
| ------ | ------ | ------------------------------------------------------------- |
| SVG    | 벡터 | 생성된 SVG 벡터 그래픽, SaveSVG 노드에 연결하여 저장 |

## 소스 코드

```python

class RecraftTextToVectorNode:
    """
    프롬프트와 해상도를 기반으로 동기적으로 SVG를 생성합니다.
    """

    RETURN_TYPES = (RecraftIO.SVG,)
    DESCRIPTION = cleandoc(__doc__ or "")  # None 값 처리
    FUNCTION = "api_call"
    API_NODE = True
    CATEGORY = "api node/image/Recraft"

    @classmethod
    def INPUT_TYPES(s):
        return {
            "required": {
                "prompt": (
                    IO.STRING,
                    {
                        "multiline": True,
                        "default": "",
                        "tooltip": "이미지 생성을 위한 프롬프트.",
                    },
                ),
                "substyle": (get_v3_substyles(RecraftStyleV3.vector_illustration),),
                "size": (
                    [res.value for res in RecraftImageSize],
                    {
                        "default": RecraftImageSize.res_1024x1024,
                        "tooltip": "생성된 이미지의 크기.",
                    },
                ),
                "n": (
                    IO.INT,
                    {
                        "default": 1,
                        "min": 1,
                        "max": 6,
                        "tooltip": "생성할 이미지 수.",
                    },
                ),
                "seed": (
                    IO.INT,
                    {
                        "default": 0,
                        "min": 0,
                        "max": 0xFFFFFFFFFFFFFFFF,
                        "control_after_generate": True,
                        "tooltip": "노드가 다시 실행될지 여부를 결정하는 시드; 실제 결과는 시드와 관계없이 비결정적입니다.",
                    },
                ),
            },
            "optional": {
                "negative_prompt": (
                    IO.STRING,
                    {
                        "default": "",
                        "forceInput": True,
                        "tooltip": "이미지에서 원하지 않는 요소에 대한 선택적 텍스트 설명.",
                    },
                ),
                "recraft_controls": (
                    RecraftIO.CONTROLS,
                    {
                        "tooltip": "Recraft Controls 노드를 통해 선택적 추가 제어."
                    },
                ),
            },
            "hidden": {
                "auth_token": "AUTH_TOKEN_COMFY_ORG",
            },
        }

    def api_call(
        self,
        prompt: str,
        substyle: str,
        size: str,
        n: int,
        seed,
        negative_prompt: str = None,
        recraft_controls: RecraftControls = None,
        auth_token=None,
        **kwargs,
    ):
        # RecraftStyle을 생성해 문자열이 적절히 포맷되도록 합니다 (예: "None"은 None으로 변환됨)
        recraft_style = RecraftStyle(RecraftStyleV3.vector_illustration, substyle=substyle)

        controls_api = None
        if recraft_controls:
            controls_api = recraft_controls.create_api_model()

        if not negative_prompt:
            negative_prompt = None

        operation = SynchronousOperation(
            endpoint=ApiEndpoint(
                path="/proxy/recraft/image_generation",
                method=HttpMethod.POST,
                request_model=RecraftImageGenerationRequest,
                response_model=RecraftImageGenerationResponse,
            ),
            request=RecraftImageGenerationRequest(
                prompt=prompt,
                negative_prompt=negative_prompt,
                model=RecraftModel.recraftv3,
                size=size,
                n=n,
                style=recraft_style.style,
                substyle=recraft_style.substyle,
                controls=controls_api,
            ),
            auth_token=auth_token,
        )
        response: RecraftImageGenerationResponse = operation.execute()
        svg_data = []
        for data in response.data:
            svg_data.append(download_url_to_bytesio(data.url, timeout=1024))

        return (SVG(svg_data),)
```
