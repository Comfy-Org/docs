![ComfyUI 기본 Recraft 이미지 인페인팅 노드](/images/built-in-nodes/api_nodes/recraft/recraft-image-inpainting.jpg)

Recraft 이미지 인페인팅 노드를 사용하면 이미지의 특정 영역만 수정하고 나머지는 그대로 유지할 수 있습니다. 이미지, 마스크 및 텍스트 프롬프트를 제공하면 선택한 영역을 채울 새로운 콘텐츠를 생성할 수 있습니다.

## 매개변수

### 기본 매개변수

| 매개변수 | 유형    | 기본값 | 설명                                     |
| --------- | ------- | ------- | ----------------------------------------------- |
| 이미지     | 이미지   | -       | 수정할 입력 이미지                           |
| 마스크      | 마스크    | -       | 선택한 영역을 정의하는 흑백 마스크         |
| 프롬프트    | 문자열  | ""      | 마스크된 영역에 생성할 내용을 설명하는 텍스트 |
| n         | 정수    | 1       | 생성할 결과의 개수 (1~6)                     |
| 시드      | 정수    | 0       | 난수 발생 시드 값                               |

### 옵션 매개변수

| 매개변수        | 유형             | 설명                            |
| ---------------- | ---------------- | -------------------------------------- |
| recraft_style    | Recraft 스타일    | 생성된 콘텐츠의 스타일 설정       |
| negative_prompt  | 문자열           | 생성된 콘텐츠에서 피해야 할 요소들 |
| recraft_controls | Recraft 컨트롤    | 색상과 같은 추가 컨트롤          |

### 출력

| 출력 | 유형  | 설명           |
| ------ | ----- | ----------------- |
| 이미지 | 이미지 | 수정된 이미지 결과 |


## 소스 코드

[노드 소스 코드 (2025-05-03 업데이트)]

```python
class RecraftImageInpaintingNode:
    """
    프롬프트와 마스크에 따라 이미지를 수정합니다.
    """

    RETURN_TYPES = (IO.IMAGE,)
    DESCRIPTION = cleandoc(__doc__ or "")  # None 값 처리
    FUNCTION = "api_call"
    API_NODE = True
    CATEGORY = "api node/image/Recraft"

    @classmethod
    def INPUT_TYPES(s):
        return {
            "required": {
                "image": (IO.IMAGE, ),
                "mask": (IO.MASK, ),
                "prompt": (
                    IO.STRING,
                    {
                        "multiline": True,
                        "default": "",
                        "tooltip": "이미지 생성을 위한 프롬프트.",
                    },
                ),
                "n": (
                    IO.INT,
                    {
                        "default": 1,
                        "min": 1,
                        "max": 6,
                        "tooltip": "생성할 이미지의 개수.",
                    },
                ),
                "seed": (
                    IO.INT,
                    {
                        "default": 0,
                        "min": 0,
                        "max": 0xFFFFFFFFFFFFFFFF,
                        "control_after_generate": True,
                        "tooltip": "노드가 다시 실행될지 여부를 결정하는 시드; 실제 결과는 시드와 관계없이 비결정적입니다.",
                    },
                ),
            },
            "optional": {
                "recraft_style": (RecraftIO.STYLEV3,),
                "negative_prompt": (
                    IO.STRING,
                    {
                        "default": "",
                        "forceInput": True,
                        "tooltip": "이미지에서 원치 않는 요소를 설명하는 선택적 텍스트.",
                    },
                ),
            },
            "hidden": {
                "auth_token": "AUTH_TOKEN_COMFY_ORG",
            },
        }

    def api_call(
        self,
        image: torch.Tensor,
        mask: torch.Tensor,
        prompt: str,
        n: int,
        seed,
        auth_token=None,
        recraft_style: RecraftStyle = None,
        negative_prompt: str = None,
        **kwargs,
    ):
        default_style = RecraftStyle(RecraftStyleV3.realistic_image)
        if recraft_style is None:
            recraft_style = default_style

        if not negative_prompt:
            negative_prompt = None

        request = RecraftImageGenerationRequest(
            prompt=prompt,
            negative_prompt=negative_prompt,
            model=RecraftModel.recraftv3,
            n=n,
            style=recraft_style.style,
            substyle=recraft_style.substyle,
            style_id=recraft_style.style_id,
            random_seed=seed,
        )

        # 마스크 텐서 준비
        _, H, W, _ = image.shape
        mask = mask.unsqueeze(-1)
        mask = mask.movedim(-1,1)
        mask = common_upscale(mask, width=W, height=H, upscale_method="nearest-exact", crop="disabled")
        mask = mask.movedim(1,-1)
        mask = (mask > 0.5).float()

        images = []
        total = image.shape[0]
        pbar = ProgressBar(total)
        for i in range(total):
            sub_bytes = handle_recraft_file_request(
                image=image[i],
                mask=mask[i:i+1],
                path="/proxy/recraft/images/inpaint",
                request=request,
                auth_token=auth_token,
            )
            images.append(torch.cat([bytesio_to_image_tensor(x) for x in sub_bytes], dim=0))
            pbar.update(1)

        images_tensor = torch.cat(images, dim=0)
        return (images_tensor, )
```
