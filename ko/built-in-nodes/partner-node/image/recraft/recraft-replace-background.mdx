![ComfyUI 기본 Recraft 배경 교체 노드](/images/built-in-nodes/api_nodes/recraft/recraft-replace-background.jpg)

Recraft 배경 교체 노드는 Recraft의 API를 사용해 이미지 내 피사체를 지능적으로 감지하고 텍스트 프롬프트에 기반한 새로운 배경을 생성합니다.

## 매개변수

### 기본 매개변수

| 매개변수 | 유형    | 기본값 | 설명                           |
| --------- | ------- | ------- | ------------------------------------- |
| image     | image   | -       | 보존할 피사체가 포함된 입력 이미지  |
| prompt    | string  | ""      | 배경 생성을 위한 텍스트 프롬프트    |
| n         | integer | 1       | 생성할 이미지 수 (1~6)              |
| seed      | integer | 0       | 노드 재실행 시 사용할 난수 시드 값    |

### 선택적 매개변수

| 매개변수       | 유형          | 설명                              |
| --------------- | ------------- | ---------------------------------------- |
| recraft_style   | Recraft 스타일 | 배경 생성을 위한 스타일 설정        |
| negative_prompt | string        | 피해야 할 요소를 설명하는 텍스트    |

### 출력

| 출력 | 유형  | 설명                          |
| ------ | ----- | ------------------------------------ |
| IMAGE  | image | 배경이 교체된 최종 이미지        |


## 소스 코드

[노드 소스 코드 (2025-05-03 업데이트)]

```python

class RecraftReplaceBackgroundNode:
    """
    제공된 프롬프트를 기반으로 이미지의 배경을 교체합니다.
    """

    RETURN_TYPES = (IO.IMAGE,)
    DESCRIPTION = cleandoc(__doc__ or "")  # None 값 처리
    FUNCTION = "api_call"
    API_NODE = True
    CATEGORY = "api node/image/Recraft"

    @classmethod
    def INPUT_TYPES(s):
        return {
            "required": {
                "image": (IO.IMAGE, ),
                "prompt": (
                    IO.STRING,
                    {
                        "multiline": True,
                        "default": "",
                        "tooltip": "이미지 생성을 위한 프롬프트.",
                    },
                ),
                "n": (
                    IO.INT,
                    {
                        "default": 1,
                        "min": 1,
                        "max": 6,
                        "tooltip": "생성할 이미지 수.",
                    },
                ),
                "seed": (
                    IO.INT,
                    {
                        "default": 0,
                        "min": 0,
                        "max": 0xFFFFFFFFFFFFFFFF,
                        "control_after_generate": True,
                        "tooltip": "노드를 다시 실행할지 결정하는 시드; 실제 결과는 시드와 관계없이 비결정적입니다.",
                    },
                ),
            },
            "optional": {
                "recraft_style": (RecraftIO.STYLEV3,),
                "negative_prompt": (
                    IO.STRING,
                    {
                        "default": "",
                        "forceInput": True,
                        "tooltip": "이미지에서 원하지 않는 요소를 설명하는 선택적 텍스트.",
                    },
                ),
            },
            "hidden": {
                "auth_token": "AUTH_TOKEN_COMFY_ORG",
            },
        }

    def api_call(
        self,
        image: torch.Tensor,
        prompt: str,
        n: int,
        seed,
        auth_token=None,
        recraft_style: RecraftStyle = None,
        negative_prompt: str = None,
        **kwargs,
    ):
        default_style = RecraftStyle(RecraftStyleV3.realistic_image)
        if recraft_style is None:
            recraft_style = default_style

        if not negative_prompt:
            negative_prompt = None

        request = RecraftImageGenerationRequest(
            prompt=prompt,
            negative_prompt=negative_prompt,
            model=RecraftModel.recraftv3,
            n=n,
            style=recraft_style.style,
            substyle=recraft_style.substyle,
            style_id=recraft_style.style_id,
        )

        images = []
        total = image.shape[0]
        pbar = ProgressBar(total)
        for i in range(total):
            sub_bytes = handle_recraft_file_request(
                image=image[i],
                path="/proxy/recraft/images/replaceBackground",
                request=request,
                auth_token=auth_token,
            )
            images.append(torch.cat([bytesio_to_image_tensor(x) for x in sub_bytes], dim=0))
            pbar.update(1)

        images_tensor = torch.cat(images, dim=0)
        return (images_tensor, )
```
