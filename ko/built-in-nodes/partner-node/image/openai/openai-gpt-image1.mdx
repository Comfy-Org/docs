![ComfyUI 기본 OpenAI GPT 이미지 1 노드](/images/built-in-nodes/api_nodes/openai/openai-gpt-image-1.jpg)

이 노드는 OpenAI의 GPT 이미지 1 API에 연결되어, 사용자가 상세한 텍스트 프롬프트를 통해 이미지를 생성할 수 있도록 합니다. 기존 DALL·E 모델과 달리, GPT 이미지 1은 GPT-4의 언어 이해 능력을 활용해 보다 복잡한 프롬프트를 처리하고 사용자의 의도에 더 잘 맞는 이미지를 생성합니다.

## 매개변수

### 기본 매개변수

| 매개변수 | 유형      | 기본값 | 설명                                                               |
| --------- | --------- | ------- | -------------------------------------------------------------------- |
| 프롬프트    | 문자열    | ""      | 생성할 내용을 설명하는 텍스트 프롬프트                             |
| 품질       | 선택사항  | "낮음"  | 이미지 품질 수준, 옵션: "낮음", "중간", "높음"                     |
| 크기       | 선택사항  | "자동"  | 출력 이미지 크기, 옵션: "자동", "1024x1024", "1024x1536", "1536x1024" |

### 이미지 편집 매개변수

| 매개변수 | 유형  | 설명                                                  |
| --------- | ----- | ----------------------------------------------------- |
| 이미지     | 이미지 | 편집할 입력 이미지, 여러 이미지 지원                  |
| 마스크      | 마스크  | 변경할 영역을 지정하는 선택적 마스크 (단일 이미지만 가능) |

### 선택적 매개변수

| 매개변수  | 유형      | 설명                                   |
| ---------- | --------- | -------------------------------------- |
| 배경       | 선택사항  | 배경 옵션: "불투명" 또는 "투명"         |
| 시드       | 정수      | 랜덤 시드 (백엔드에서 아직 구현되지 않음)  |
| n          | 정수      | 생성할 이미지 수, 범위 1~8              |

### 출력

| 출력 | 유형  | 설명            |
| ------ | ----- | ---------------- |
| 이미지  | 이미지 | 생성된 이미지 결과 |

## 작동 방식

OpenAI GPT 이미지 1 노드는 GPT-4의 언어 이해 능력과 이미지 생성을 결합합니다. 텍스트 프롬프트를 분석해 의미와 의도를 파악한 후, 이에 맞는 이미지를 생성합니다.

이미지 편집 모드에서는 기존 이미지를 수정할 수 있습니다. 마스크를 사용하면 어떤 영역을 정밀하게 변경할지 제어할 수 있습니다. 단, 마스크 입력은 단일 이미지 입력에서만 작동합니다.

사용자는 품질 수준, 크기, 배경 처리, 생성 수 등 다양한 매개변수를 조정해 출력을 제어할 수 있습니다.

```python

class OpenAIGPTImage1(ComfyNodeABC):
    """
    OpenAI의 GPT 이미지 1 엔드포인트를 통해 동기적으로 이미지를 생성합니다.

    /proxy/openai/images/generations 경유로 요청합니다. 반환된 URL은 단기 유효하므로, 결과를 유지해야 한다면 다운로드하거나 캐시하세요.
    """

    def __init__(self):
        pass

    @classmethod
    def INPUT_TYPES(cls) -> InputTypeDict:
        return {
            "required": {
                "프롬프트": (
                    IO.STRING,
                    {
                        "multiline": True,
                        "default": "",
                        "tooltip": "GPT 이미지 1용 텍스트 프롬프트",
                    },
                ),
            },
            "optional": {
                "시드": (
                    IO.INT,
                    {
                        "default": 0,
                        "min": 0,
                        "max": 2**31 - 1,
                        "step": 1,
                        "display": "number",
                        "control_after_generate": True,
                        "tooltip": "백엔드에서 아직 구현되지 않음",
                    },
                ),
                "품질": (
                    IO.COMBO,
                    {
                        "options": ["낮음", "중간", "높음"],
                        "default": "낮음",
                        "tooltip": "이미지 품질, 비용과 생성 시간에 영향을 줍니다.",
                    },
                ),
                "배경": (
                    IO.COMBO,
                    {
                        "options": ["불투명", "투명"],
                        "default": "불투명",
                        "tooltip": "배경 포함 또는 제외한 이미지 반환",
                    },
                ),
                "크기": (
                    IO.COMBO,
                    {
                        "options": ["자동", "1024x1024", "1024x1536", "1536x1024"],
                        "default": "자동",
                        "tooltip": "이미지 크기",
                    },
                ),
                "n": (
                    IO.INT,
                    {
                        "default": 1,
                        "min": 1,
                        "max": 8,
                        "step": 1,
                        "display": "number",
                        "tooltip": "생성할 이미지 수",
                    },
                ),
                "이미지": (
                    IO.IMAGE,
                    {
                        "default": None,
                        "tooltip": "이미지 편집용 선택적 참조 이미지",
                    },
                ),
                "마스크": (
                    IO.MASK,
                    {
                        "default": None,
                        "tooltip": "선택적 마스크로 인페인팅 가능 (흰색 영역이 교체됨)",
                    },
                ),
            },
            "hidden": {"auth_token": "AUTH_TOKEN_COMFY_ORG"},
        }

    RETURN_TYPES = (IO.IMAGE,)
    FUNCTION = "api_call"
    CATEGORY = "api node/image/openai"
    DESCRIPTION = cleandoc(__doc__ or "")
    API_NODE = True

    def api_call(
        self,
        프롬프트,
        시드=0,
        품질="낮음",
        배경="불투명",
        이미지=None,
        마스크=None,
        n=1,
        크기="1024x1024",
        auth_token=None,
    ):
        model = "gpt-image-1"
        path = "/proxy/openai/images/generations"
        content_type="application/json"
        request_class = OpenAIImageGenerationRequest
        img_binaries = []
        mask_binary = None
        files = []

        if 이미지 is not None:
            path = "/proxy/openai/images/edits"
            request_class = OpenAIImageEditRequest
            content_type ="multipart/form-data"

            batch_size = 이미지.shape[0]

            for i in range(batch_size):
                single_image = 이미지[i : i + 1]
                scaled_image = downscale_image_tensor(single_image).squeeze()

                image_np = (scaled_image.numpy() * 255).astype(np.uint8)
                img = Image.fromarray(image_np)
                img_byte_arr = io.BytesIO()
                img.save(img_byte_arr, format="PNG")
                img_byte_arr.seek(0)
                img_binary = img_byte_arr
                img_binary.name = f"image_{i}.png"

                img_binaries.append(img_binary)
                if batch_size == 1:
                    files.append(("image", img_binary))
                else:
                    files.append(("image[]", img_binary))

        if 마스크 is not None:
            if 이미지.shape[0] != 1:
                raise Exception("여러 이미지에 마스크를 사용할 수 없습니다")
            if 이미지 is None:
                raise Exception("입력 이미지 없이 마스크를 사용할 수 없습니다")
            if 마스크.shape[1:] != 이미지.shape[1:-1]:
                raise Exception("마스크와 이미지는 같은 크기여야 합니다")
            batch, height, width = 마스크.shape
            rgba_mask = torch.zeros(height, width, 4, device="cpu")
            rgba_mask[:, :, 3] = 1 - 마스크.squeeze().cpu()

            scaled_mask = downscale_image_tensor(rgba_mask.unsqueeze(0)).squeeze()

            mask_np = (scaled_mask.numpy() * 255).astype(np.uint8)
            mask_img = Image.fromarray(mask_np)
            mask_img_byte_arr = io.BytesIO()
            mask_img.save(mask_img_byte_arr, format="PNG")
            mask_img_byte_arr.seek(0)
            mask_binary = mask_img_byte_arr
            mask_binary.name = "mask.png"
            files.append(("mask", mask_binary))

        # 작업 구성
        operation = SynchronousOperation(
            endpoint=ApiEndpoint(
                path=path,
                method=HttpMethod.POST,
                request_model=request_class,
                response_model=OpenAIImageGenerationResponse,
            ),
            request=request_class(
                model=model,
                prompt=프롬프트,
                quality=품질,
                background=배경,
                n=n,
                seed=시드,
                size=크기,
            ),
            files=files if files else None,
            content_type=content_type,
            auth_token=auth_token,
        )

        response = operation.execute()

        img_tensor = validate_and_cast_response(response)
        return (img_tensor,)
``` 
