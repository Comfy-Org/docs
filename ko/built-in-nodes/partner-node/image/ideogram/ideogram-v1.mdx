![ComfyUI 네이티브 이디오그램 V1 노드](/images/built-in-nodes/api_nodes/ideogram/ideogram-v1.jpg)

이디오그램 V1 노드는 이디오그램의 텍스트-to-이미지 API를 사용해 고품질 텍스트 렌더링 기능을 갖춘 이미지를 생성할 수 있게 해줍니다.

## 매개변수 설명

### 필수 매개변수

| 매개변수           | 유형    | 기본값 | 설명                                                                 |
| ------------------- | ------- | ------- | --------------------------------------------------------------------------- |
| 프롬프트              | 문자열  | ""      | 생성할 내용을 설명하는 텍스트 프롬프트                              |
| 터보               | 부울    | False   | 터보 모드 사용 여부 (더 빠르지만 품질이 다소 낮을 수 있음)               |
| 종횡비                | 선택    | "1:1"   | 이미지 종횡비                                                          |
| 마법 프롬프트 옵션     | 선택    | "AUTO"  | 생성 시 마법 프롬프트 사용 여부 결정, 옵션: AUTO, ON, OFF         |
| 시드                | 정수    | 0       | 임의 시드 값 (0~2147483647)                                            |
| 부정적 프롬프트      | 문자열  | ""      | 이미지에서 제외하고 싶은 요소를 지정                               |
| 이미지 개수          | 정수    | 1       | 생성할 이미지 개수 (1~8)                                                |


### 출력

| 출력 | 유형  | 설명            |
| ------ | ----- | ---------------------- |
| 이미지  | 이미지 | 생성된 이미지 결과 |

## 소스 코드

[노드 소스 코드 (2025-05-03 업데이트)] 

```python
class IdeogramV1(ComfyNodeABC):
    """
    이디오그램 V1 모델을 사용해 동기적으로 이미지를 생성합니다.

    이미지 링크는 일정 기간만 제공되며, 이미지를 보관하려면 다운로드해야 합니다.
    """

    def __init__(self):
        pass

    @classmethod
    def INPUT_TYPES(cls) -> InputTypeDict:
        return {
            "required": {
                "prompt": (
                    IO.STRING,
                    {
                        "multiline": True,
                        "default": "",
                        "tooltip": "이미지 생성을 위한 프롬프트",
                    },
                ),
                "turbo": (
                    IO.BOOLEAN,
                    {
                        "default": False,
                        "tooltip": "터보 모드 사용 여부 (더 빠른 생성, 품질이 다소 낮을 수 있음)",
                    }
                ),
            },
            "optional": {
                "aspect_ratio": (
                    IO.COMBO,
                    {
                        "options": list(V1_V2_RATIO_MAP.keys()),
                        "default": "1:1",
                        "tooltip": "이미지 생성을 위한 종횡비.",
                    },
                ),
                "magic_prompt_option": (
                    IO.COMBO,
                    {
                        "options": ["AUTO", "ON", "OFF"],
                        "default": "AUTO",
                        "tooltip": "생성 시 마법 프롬프트 사용 여부 결정",
                    },
                ),
                "seed": (
                    IO.INT,
                    {
                        "default": 0,
                        "min": 0,
                        "max": 2147483647,
                        "step": 1,
                        "control_after_generate": True,
                        "display": "number",
                    },
                ),
                "negative_prompt": (
                    IO.STRING,
                    {
                        "multiline": True,
                        "default": "",
                        "tooltip": "이미지에서 제외할 내용 설명",
                    },
                ),
                "num_images": (
                    IO.INT,
                    {"default": 1, "min": 1, "max": 8, "step": 1, "display": "number"},
                ),
            },
            "hidden": {"auth_token": "AUTH_TOKEN_COMFY_ORG"},
        }

    RETURN_TYPES = (IO.IMAGE,)
    FUNCTION = "api_call"
    CATEGORY = "api node/image/ideogram/v1"
    DESCRIPTION = cleandoc(__doc__ or "")
    API_NODE = True

    def api_call(
        self,
        prompt,
        turbo=False,
        aspect_ratio="1:1",
        magic_prompt_option="AUTO",
        seed=0,
        negative_prompt="",
        num_images=1,
        auth_token=None,
    ):
        # 터보 설정에 따라 모델 결정
        aspect_ratio = V1_V2_RATIO_MAP.get(aspect_ratio, None)
        model = "V_1_TURBO" if turbo else "V_1"

        operation = SynchronousOperation(
            endpoint=ApiEndpoint(
                path="/proxy/ideogram/generate",
                method=HttpMethod.POST,
                request_model=IdeogramGenerateRequest,
                response_model=IdeogramGenerateResponse,
            ),
            request=IdeogramGenerateRequest(
                image_request=ImageRequest(
                    prompt=prompt,
                    model=model,
                    num_images=num_images,
                    seed=seed,
                    aspect_ratio=aspect_ratio if aspect_ratio != "ASPECT_1_1" else None,
                    magic_prompt_option=(
                        magic_prompt_option if magic_prompt_option != "AUTO" else None
                    ),
                    negative_prompt=negative_prompt if negative_prompt else None,
                )
            ),
            auth_token=auth_token,
        )

        response = operation.execute()

        if not response.data or len(response.data) == 0:
            raise Exception("응답에서 이미지가 생성되지 않았습니다")

        image_urls = [image_data.url for image_data in response.data if image_data.url]

        if not image_urls:
            raise Exception("응답에서 이미지 URL이 생성되지 않았습니다")

        return (download_and_process_images(image_urls),)
```
