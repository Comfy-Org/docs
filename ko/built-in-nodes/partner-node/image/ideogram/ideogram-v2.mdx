![ComfyUI 내장 이디오그램 V2 노드](/images/built-in-nodes/api_nodes/ideogram/ideogram-v2.jpg)

이디오그램 V2 노드를 사용하면 이디오그램의 2세대 AI 모델을 활용해 더욱 정교한 이미지를 생성할 수 있으며, 텍스트 렌더링, 이미지 품질 및 전반적인 미학이 크게 개선되었습니다.

## 매개변수

### 필수 매개변수

| 매개변수 | 유형    | 기본값 | 설명                                                           |
| --------- | ------- | ------- | --------------------------------------------------------------------- |
| 프롬프트    | 문자열  | ""      | 생성할 콘텐츠를 설명하는 텍스트 프롬프트                        |
| 터보     | 부울    | False   | 터보 모드 사용 여부 (더 빠른 생성, 잠재적으로 낮은 품질) |

### 선택적 매개변수

| 매개변수           | 유형     | 기본값 | 설명                                                                                                   |
| ------------------- | -------- | ------- | ------------------------------------------------------------------------------------------------------------- |
| 종횡비        | 드롭다운 | "1:1"   | 이미지 종횡비, 해상도가 "자동"으로 설정된 경우에 유효합니다                                |
| 해상도          | 드롭다운 | "자동"  | 출력 이미지 해상도, "자동"으로 설정되지 않은 경우 종횡비 설정을 무시합니다                      |
| 마법 프롬프트 옵션 | 드롭다운 | "AUTO"  | 생성 시 마법 프롬프트 기능 사용 여부 결정, 옵션은 ["AUTO", "ON", "OFF"]            |
| 시드                | 정수    | 0       | 난수 시드 값, 범위 0-2147483647                                                                         |
| 스타일 유형          | 드롭다운 | "NONE"  | 생성 스타일 유형 (V2만 해당), 옵션은 ["AUTO", "GENERAL", "REALISTIC", "DESIGN", "RENDER_3D", "ANIME"] |
| 부정적 프롬프트     | 문자열   | ""      | 이미지에 나타나지 않기를 원하는 요소를 지정합니다                                                      |
| 이미지 수          | 정수    | 1       | 생성할 이미지 수, 범위 1-8                                                                       |

### 출력

| 출력 | 유형  | 설명        |
| ------ | ----- | ------------------ |
| 이미지  | 이미지 | 생성된 이미지(s) |

## 소스 코드

[노드 소스 코드 (2025-05-03 업데이트)]
```python

class IdeogramV2(ComfyNodeABC):
    """
    이디오그램 V2 모델을 사용해 동기식으로 이미지를 생성합니다.

    이미지 링크는 일정 기간 동안만 제공되며, 이미지를 유지하고 싶다면 다운로드해야 합니다.
    """

    def __init__(self):
        pass

    @classmethod
    def INPUT_TYPES(cls) -> InputTypeDict:
        return {
            "required": {
                "prompt": (
                    IO.STRING,
                    {
                        "multiline": True,
                        "default": "",
                        "tooltip": "이미지 생성을 위한 프롬프트",
                    },
                ),
                "turbo": (
                    IO.BOOLEAN,
                    {
                        "default": False,
                        "tooltip": "터보 모드 사용 여부 (더 빠른 생성, 잠재적으로 낮은 품질)",
                    }
                ),
            },
            "optional": {
                "aspect_ratio": (
                    IO.COMBO,
                    {
                        "options": list(V1_V2_RATIO_MAP.keys()),
                        "default": "1:1",
                        "tooltip": "이미지 생성을 위한 종횡비. 해상도가 AUTO로 설정되지 않은 경우 무시됩니다.",
                    },
                ),
                "resolution": (
                    IO.COMBO,
                    {
                        "options": list(V1_V1_RES_MAP.keys()),
                        "default": "Auto",
                        "tooltip": "이미지 생성을 위한 해상도. AUTO로 설정되지 않은 경우 종횡비 설정을 무시합니다.",
                    },
                ),
                "magic_prompt_option": (
                    IO.COMBO,
                    {
                        "options": ["AUTO", "ON", "OFF"],
                        "default": "AUTO",
                        "tooltip": "생성 시 마법 프롬프트 사용 여부 결정",
                    },
                ),
                "seed": (
                    IO.INT,
                    {
                        "default": 0,
                        "min": 0,
                        "max": 2147483647,
                        "step": 1,
                        "control_after_generate": True,
                        "display": "number",
                    },
                ),
                "style_type": (
                    IO.COMBO,
                    {
                        "options": ["AUTO", "GENERAL", "REALISTIC", "DESIGN", "RENDER_3D", "ANIME"],
                        "default": "NONE",
                        "tooltip": "생성 스타일 유형 (V2만 해당)",
                    },
                ),
                "negative_prompt": (
                    IO.STRING,
                    {
                        "multiline": True,
                        "default": "",
                        "tooltip": "이미지에서 제외할 내용 설명",
                    },
                ),
                "num_images": (
                    IO.INT,
                    {"default": 1, "min": 1, "max": 8, "step": 1, "display": "number"},
                ),
                #"color_palette": (
                #    IO.STRING,
                #    {
                #        "multiline": False,
                #        "default": "",
                #        "tooltip": "컬러 팔레트 사전 설정 이름 또는 가중치가 있는 16진수 색상",
                #    },
                #),
            },
            "hidden": {"auth_token": "AUTH_TOKEN_COMFY_ORG"},
        }

    RETURN_TYPES = (IO.IMAGE,)
    FUNCTION = "api_call"
    CATEGORY = "api node/image/ideogram/v2"
    DESCRIPTION = cleandoc(__doc__ or "")
    API_NODE = True

    def api_call(
        self,
        prompt,
        turbo=False,
        aspect_ratio="1:1",
        resolution="Auto",
        magic_prompt_option="AUTO",
        seed=0,
        style_type="NONE",
        negative_prompt="",
        num_images=1,
        color_palette="",
        auth_token=None,
    ):
        aspect_ratio = V1_V2_RATIO_MAP.get(aspect_ratio, None)
        resolution = V1_V1_RES_MAP.get(resolution, None)
        # 터보 설정에 따라 모델 결정
        model = "V_2_TURBO" if turbo else "V_2"

        # 해상도 vs 종횡비 로직 처리
        # 해상도가 AUTO가 아닌 경우 종횡비를 무시합니다
        final_resolution = None
        final_aspect_ratio = None

        if resolution != "AUTO":
            final_resolution = resolution
        else:
            final_aspect_ratio = aspect_ratio if aspect_ratio != "ASPECT_1_1" else None

        operation = SynchronousOperation(
            endpoint=ApiEndpoint(
                path="/proxy/ideogram/generate",
                method=HttpMethod.POST,
                request_model=IdeogramGenerateRequest,
                response_model=IdeogramGenerateResponse,
            ),
            request=IdeogramGenerateRequest(
                image_request=ImageRequest(
                    prompt=prompt,
                    model=model,
                    num_images=num_images,
                    seed=seed,
                    aspect_ratio=final_aspect_ratio,
                    resolution=final_resolution,
                    magic_prompt_option=(
                        magic_prompt_option if magic_prompt_option != "AUTO" else None
                    ),
                    style_type=style_type if style_type != "NONE" else None,
                    negative_prompt=negative_prompt if negative_prompt else None,
                    color_palette=color_palette if color_palette else None,
                )
            ),
            auth_token=auth_token,
        )

        response = operation.execute()

        if not response.data or len(response.data) == 0:
            raise Exception("응답에서 생성된 이미지가 없습니다")

        image_urls = [image_data.url for image_data in response.data if image_data.url]

        if not image_urls:
            raise Exception("응답에서 생성된 이미지 URL이 없습니다")

        return (download_and_process_images(image_urls),)
```
