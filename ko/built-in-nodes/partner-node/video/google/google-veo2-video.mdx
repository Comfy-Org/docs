Google Veo2 비디오 노드는 Google의 Veo2 API 기술을 사용해 텍스트 설명으로 고품질 비디오를 생성하며, 텍스트 프롬프트를 동적 비디오 콘텐츠로 변환합니다.

## 매개변수

### 기본 매개변수

| 매개변수         | 유형    | 기본값 | 설명                                                |
| ---------------- | ------- | ------- | ---------------------------------------------------------- |
| 프롬프트           | 문자열  | ""      | 생성할 비디오 콘텐츠의 텍스트 설명          |
| 종횡비             | 선택    | "16:9"  | 출력 비디오 종횡비, "16:9" 또는 "9:16"               |
| 부정적 프롬프트     | 문자열  | ""      | 비디오에서 피해야 할 내용을 설명한 텍스트       |
| 지속 시간 초        | 정수    | 5       | 비디오 지속 시간, 5~8초                                |
| 프롬프트 강화       | 불리언  | True    | AI를 사용해 프롬프트를 강화할지 여부                    |
| 인물 생성 허용     | 선택    | "허용"  | 인물 생성을 허용하거나 차단할지, "허용" 또는 "차단"       |
| 시드              | 정수    | 0       | 랜덤 시드, 0은 무작위로 생성됨                    |

### 옵션 매개변수

| 매개변수 | 유형  | 기본값 | 설명                                    |
| --------- | ----- | ------- | ---------------------------------------------- |
| 이미지     | 이미지 | 없음    | 비디오 생성을 안내할 수 있는 선택적 참조 이미지|

### 출력

| 출력 | 유형  | 설명      |
| ------ | ----- | ---------------- |
| 비디오  | 비디오 | 생성된 비디오  |

class VeoVideoGenerationNode(ComfyNodeABC):
    """
    Google의 Veo API를 사용해 텍스트 프롬프트로 비디오를 생성합니다.

    이 노드는 텍스트 설명과 선택적 이미지 입력을 통해 비디오를 만들 수 있으며,
    종횡비, 지속 시간 등 다양한 매개변수를 제어할 수 있습니다.
    """

    @classmethod

        return {
            "required": {

                    {
                        "options": ["16:9", "9:16"],
                        "default": "16:9",

                    },
                ),
            },
            "optional": {
                "부정적 프롬프트": (
                    IO.STRING,
                    {
                        "multiline": True,
                        "default": "",
                        "tooltip": "비디오에서 피해야 할 내용을 안내하는 부정적 텍스트 프롬프트",
                    },
                ),
                "지속 시간 초": (
                    IO.INT,
                    {
                        "default": 5,
                        "min": 5,
                        "max": 8,
                        "step": 1,
                        "display": "number",
                        "tooltip": "출력 비디오의 지속 시간 (초)",
                    },
                ),
                "프롬프트 강화": (
                    IO.BOOLEAN,
                    {
                        "default": True,
                        "tooltip": "AI를 사용해 프롬프트를 강화할지 여부",
                    }
                ),
                "인물 생성 허용": (
                    IO.COMBO,
                    {
                        "options": ["허용", "차단"],
                        "default": "허용",
                        "tooltip": "비디오에 인물을 생성하도록 허용할지 여부",
                    },
                ),
                "시드": (
                    IO.INT,
                    {
                        "default": 0,
                        "min": 0,
                        "max": 0xFFFFFFFF,
                        "step": 1,
                        "display": "number",
                        "control_after_generate": True,
                        "tooltip": "비디오 생성을 위한 시드 (0은 무작위)",
                    },
                ),
                "이미지": (IO.IMAGE, {
                    "default": None,
                    "tooltip": "비디오 생성을 안내할 수 있는 선택적 참조 이미지",
                }),
            },
            "hidden": {
                "auth_token": "AUTH_TOKEN_COMFY_ORG",
            },
        }

    RETURN_TYPES = (IO.VIDEO,)
    FUNCTION = "generate_video"
    CATEGORY = "api node/video/Veo"
    DESCRIPTION = "Google의 Veo API를 사용해 텍스트 프롬프트로 비디오를 생성합니다"
    API_NODE = True

    def generate_video(
        self,
        프롬프트,
        종횡비="16:9",
        부정적 프롬프트="",
        지속 시간 초=5,
        프롬프트 강화=True,
        인물 생성 허용="허용",
        시드=0,
        이미지=None,
        auth_token=None,
    ):
        # 요청을 위한 인스턴스 준비
        인스턴스 = []

        instance = {
            "프롬프트": 프롬프트
        }

        # 이미지가 제공되면 추가
        if 이미지 is not None:
            image_base64 = convert_image_to_base64(이미지)
            if image_base64:
                instance["이미지"] = {
                    "bytesBase64Encoded": image_base64,
                    "mimeType": "image/png"
                }

        인스턴스.append(instance)

        # 매개변수 딕셔너리 생성
        매개변수 = {
            "종횡비": 종횡비,
            "인물 생성 허용": 인물 생성 허용,
            "지속 시간 초": 지속 시간 초,
            "프롬프트 강화": 프롬프트 강화,
        }

        # 선택적 매개변수 추가 (만약 제공되었다면)
        if 부정적 프롬프트:
            매개변수["부정적 프롬프트"] = 부정적 프롬프트
        if 시드 > 0:
            매개변수["시드"] = 시드

        # 비디오 생성 시작을 위한 초기 요청
        초기 작업 = SynchronousOperation(
            endpoint=ApiEndpoint(
                path="/proxy/veo/generate",
                method=HttpMethod.POST,
                request_model=Veo2GenVidRequest,
                response_model=Veo2GenVidResponse
            ),
            request=Veo2GenVidRequest(
                instances=인스턴스,
                parameters=매개변수
            ),
            auth_token=auth_token
        )

        초기 응답 = 초기 작업.execute()
        작업 이름 = 초기 응답.name

        logging.info(f"Veo 생성 시작, 작업 이름: {작업 이름}")

        # 상태 추출 함수 정의
        def 상태 추출기(response):
            # 작업이 완료되었으면 "완료"만 반환, 성공 여부와 상관없이
            # polling이 완료된 후 오류를 확인합니다.
            return "완료" if response.done else "진행 중"

        # 진행률 추출 함수 정의
        def 진행률 추출기(response):
            # API에서 진행률 정보를 제공한다면 개선 가능
            return None

        # polling 작업 정의
        polling 작업 = PollingOperation(
            poll_endpoint=ApiEndpoint(
                path="/proxy/veo/poll",
                method=HttpMethod.POST,
                request_model=Veo2GenVidPollRequest,
                response_model=Veo2GenVidPollResponse
            ),
            completed_statuses=["완료"],
            failed_statuses=[],  # 실패 상태는 없으며, polling 후 오류를 처리합니다.
            status_extractor=상태 추출기,
            progress_extractor=진행률 추출기,
            request=Veo2GenVidPollRequest(
                operationName=작업 이름
            ),
            auth_token=auth_token,
            poll_interval=5.0
        )

        # polling 작업 실행
        polling 응답 = polling 작업.execute()

        # 최종 응답에서 오류 확인
        # polling 응답에 오류가 있는지 확인
        if hasattr(polling 응답, 'error') and polling 응답.error:
            error_message = f"Veo API 오류: {polling 응답.error.message} (코드: {polling 응답.error.code})"
            logging.error(error_message)
            raise Exception(error_message)

        # RAI 필터링된 콘텐츠 확인
        if (hasattr(polling 응답.response, 'raiMediaFilteredCount') and
            polling 응답.response.raiMediaFilteredCount > 0):

            # 이유 메시지가 있다면 추출
            if (hasattr(polling 응답.response, 'raiMediaFilteredReasons') and
                polling 응답.response.raiMediaFilteredReasons):
                reason = polling 응답.response.raiMediaFilteredReasons[0]
                error_message = f"Google의 책임 있는 AI 관행에 의해 콘텐츠가 필터링되었습니다: {reason} ({polling 응답.response.raiMediaFilteredCount}개의 비디오가 필터링되었습니다.)"
            else:
                error_message = f"Google의 책임 있는 AI 관행에 의해 콘텐츠가 필터링되었습니다 ({polling 응답.response.raiMediaFilteredCount}개의 비디오가 필터링되었습니다.)"

            logging.error(error_message)
            raise Exception(error_message)

        # 비디오 데이터 추출
        비디오 데이터 = None
        if polling 응답.response and hasattr(polling 응답.response, 'videos') and polling 응답.response.videos and len(polling 응답.response.videos) > 0:
            비디오 = polling 응답.response.videos[0]

            # 비디오가 base64나 URL로 제공되었는지 확인
            if hasattr(비디오, 'bytesBase64Encoded') and 비디오.bytesBase64Encoded:
                # base64 문자열을 바이트로 디코딩
                비디오 데이터 = base64.b64decode(비디오.bytesBase64Encoded)
            elif hasattr(비디오, 'gcsUri') and 비디오.gcsUri:
                # URL에서 다운로드
                비디오_url = 비디오.gcsUri
                비디오_response = requests.get(비디오_url)
                비디오 데이터 = 비디오_response.content
            else:
                raise Exception("비디오가 반환되었지만 데이터나 URL이 제공되지 않았습니다")
        else:
            raise Exception("비디오 생성이 완료되었으나 비디오가 반환되지 않았습니다")

        if not 비디오 데이터:
            raise Exception("비디오 데이터가 반환되지 않았습니다")

        logging.info("비디오 생성이 성공적으로 완료되었습니다")

        # 비디오 데이터를 BytesIO 객체로 변환
        비디오_io = io.BytesIO(비디오 데이터)

        # VideoFromFile 객체 반환
        return (VideoFromFile(비디오_io),)
