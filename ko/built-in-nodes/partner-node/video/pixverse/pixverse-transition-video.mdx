![ComfyUI 기본 PixVerse 전환 동영상 노드](/images/built-in-nodes/api_nodes/pixverse/pixverse-transition-video.jpg)

PixVerse 전환 동영상 노드는 PixVerse의 API에 연결하여 두 이미지 간의 부드러운 동영상 전환을 생성합니다. 이 노드는 모든 중간 프레임을 자동으로 생성해 유연한 변형을 만들어내며, 모핑 효과, 장면 전환 및 객체 진화에 완벽합니다.

## 매개변수

### 필수 매개변수

| 매개변수        | 타입    | 기본값                    | 설명                                 |
| ---------------- | ------- | -------------------------- | ------------------------------------------- |
| 첫 번째 프레임      | 이미지   | -                          | 시작 프레임 이미지                        |
| 마지막 프레임       | 이미지   | -                          | 끝 프레임 이미지                          |
| 프롬프트           | 문자열  | ""                         | 동영상과 전환을 설명하는 텍스트 프롬프트 |
| 품질              | 선택    | "PixverseQuality.res_540p" | 출력 동영상 품질                        |
| 지속 시간 초     | 선택    | -                          | 생성된 동영상의 길이                   |
| 모션 모드          | 선택    | -                          | 동영상 모션 스타일                          |
| 시드             | 정수    | 0                          | 난수 시드 (범위: 0-2147483647)           |

### 선택적 매개변수

| 매개변수         | 타입              | 기본값 | 설명                |
| ----------------- | ----------------- | ------- | -------------------------- |
| 네거티브 프롬프트   | 문자열            | ""      | 동영상에서 피하고 싶은 요소 |
| pixverse_template | PIXVERSE_TEMPLATE | None    | 선택적 스타일 프리셋      |

### 매개변수 제약사항

- 품질을 1080p로 설정할 경우, 모션 모드는 기본적으로 일반으로 고정되고 지속 시간 초는 5초로 고정됩니다.
- 지속 시간 초가 5초가 아닐 경우, 모션 모드는 기본적으로 일반으로 고정됩니다.

### 출력

| 출력 | 타입  | 설명     |
| ------ | ----- | --------------- |
| VIDEO  | 동영상 | 생성된 동영상 |

## 소스 코드

```python

class PixverseTransitionVideoNode(ComfyNodeABC):
    """
    프롬프트와 출력 크기에 따라 동영상을 동기식으로 생성합니다.
    """

    RETURN_TYPES = (IO.VIDEO,)
    DESCRIPTION = cleandoc(__doc__ or "")  # 잠재적인 None 값 처리
    FUNCTION = "api_call"
    API_NODE = True
    CATEGORY = "api node/video/Pixverse"

    @classmethod
    def INPUT_TYPES(s):
        return {
            "required": {
                "첫 번째 프레임": (
                    IO.IMAGE,
                ),
                "마지막 프레임": (
                    IO.IMAGE,
                ),
                "프롬프트": (
                    IO.STRING,
                    {
                        "multiline": True,
                        "default": "",
                        "tooltip": "동영상 생성을 위한 프롬프트",
                    },
                ),
                "품질": (
                    [resolution.value for resolution in PixverseQuality],
                    {
                        "default": PixverseQuality.res_540p,
                    },
                ),
                "지속 시간 초": ([dur.value for dur in PixverseDuration],),
                "모션 모드": ([mode.value for mode in PixverseMotionMode],),
                "시드": (
                    IO.INT,
                    {
                        "default": 0,
                        "min": 0,
                        "max": 2147483647,
                        "control_after_generate": True,
                        "tooltip": "동영상 생성을 위한 시드.",
                    },
                ),
            },
            "optional": {
                "네거티브 프롬프트": (
                    IO.STRING,
                    {
                        "default": "",
                        "forceInput": True,
                        "tooltip": "이미지에서 원치 않는 요소를 설명하는 선택적 텍스트.",
                    },
                ),
                "pixverse_template": (
                    PixverseIO.TEMPLATE,
                    {
                        "tooltip": "생성 스타일에 영향을 주는 선택적 템플릿, Pixverse Template 노드에서 생성됨."
                    }
                )
            },
            "hidden": {
                "auth_token": "AUTH_TOKEN_COMFY_ORG",
            },
        }

    def api_call(
        self,
        첫 번째 프레임: torch.Tensor,
        마지막 프레임: torch.Tensor,
        프롬프트: str,
        품질: str,
        지속 시간 초: int,
        모션 모드: str,
        시드,
        네거티브 프롬프트: str=None,
        pixverse_template: int=None,
        auth_token=None,
        **kwargs,
    ):
        첫 번째 프레임_id = upload_image_to_pixverse(첫 번째 프레임, auth_token=auth_token)
        마지막 프레임_id = upload_image_to_pixverse(마지막 프레임, auth_token=auth_token)

        # 1080p는 지속 시간 5초로 제한됨
        # 1080p 또는 5초가 아닌 지속 시간에서는 일반 모션 모드만 지원됨
        if 품질 == PixverseQuality.res_1080p:
            모션 모드 = PixverseMotionMode.normal
            지속 시간 초 = PixverseDuration.dur_5
        elif 지속 시간 초 != PixverseDuration.dur_5:
            모션 모드 = PixverseMotionMode.normal

        operation = SynchronousOperation(
            endpoint=ApiEndpoint(
                path="/proxy/pixverse/video/transition/generate",
                method=HttpMethod.POST,
                request_model=PixverseTransitionVideoRequest,
                response_model=PixverseVideoResponse,
            ),
            request=PixverseTransitionVideoRequest(
                첫 번째 프레임_img=첫 번째 프레임_id,
                마지막 프레임_img=마지막 프레임_id,
                프롬프트=프롬프트,
                품질=품질,
                지속 시간=지속 시간 초,
                모션 모드=모션 모드,
                네거티브 프롬프트=네거티브 프롬프트 if 네거티브 프롬프트 else None,
                template_id=pixverse_template,
                seed=시드,
            ),
            auth_token=auth_token,
        )
        response_api = operation.execute()

        if response_api.Resp is None:
            raise Exception(f"Pixverse 요청 실패: '{response_api.ErrMsg}'")

        operation = PollingOperation(
            poll_endpoint=ApiEndpoint(
                path=f"/proxy/pixverse/video/result/{response_api.Resp.video_id}",
                method=HttpMethod.GET,
                request_model=EmptyRequest,
                response_model=PixverseGenerationStatusResponse,
            ),
            completed_statuses=[PixverseStatus.successful],
            failed_statuses=[PixverseStatus.contents_moderation, PixverseStatus.failed, PixverseStatus.deleted],
            status_extractor=lambda x: x.Resp.status,
            auth_token=auth_token,
        )
        response_poll = operation.execute()

        vid_response = requests.get(response_poll.Resp.url)
        return (VideoFromFile(BytesIO(vid_response.content)),)
``` 
