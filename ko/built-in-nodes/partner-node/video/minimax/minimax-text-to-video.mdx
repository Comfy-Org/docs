MiniMax 텍스트에서 비디오로 노드는 MiniMax의 API에 연결하여 텍스트 프롬프트를 기반으로 고품질의 부드러운 비디오를 생성합니다. 다양한 스타일의 짧은 동영상 클립을 만들기 위해 여러 비디오 생성 모델을 지원합니다.

## 매개변수

### 필수 매개변수

| 매개변수   | 유형    | 기본값  | 설명                                                    |
| ----------- | ------- | -------- | -------------------------------------------------------------- |
| prompt_text | 문자열  | ""       | 비디오 생성을 안내하는 텍스트 프롬프트                   |
| model       | 선택    | "T2V-01" | 사용할 비디오 모델, 옵션은 "T2V-01" 및 "T2V-01-Director" |
| seed        | 정수    | 0        | 노이즈 생성용 랜덤 시드, 기본값은 0                |

### 출력

| 출력 | 유형  | 설명     |
| ------ | ----- | --------------- |
| VIDEO  | 비디오 | 생성된 비디오 |

## 소스 코드

```python

class MinimaxTextToVideoNode:
    """
    Minimax의 API를 사용해 프롬프트와 선택적 매개변수를 기반으로 동기적으로 비디오를 생성합니다.
    """

    @classmethod
    def INPUT_TYPES(s):
        return {
            "required": {
                "prompt_text": (
                    "STRING",
                    {
                        "multiline": True,
                        "default": "",
                        "tooltip": "비디오 생성을 안내하는 텍스트 프롬프트",
                    },
                ),
                "model": (
                    [
                        "T2V-01",
                        "T2V-01-Director",
                    ],
                    {
                        "default": "T2V-01",
                        "tooltip": "비디오 생성에 사용할 모델",
                    },
                ),
            },
            "optional": {
                "seed": (
                    IO.INT,
                    {
                        "default": 0,
                        "min": 0,
                        "max": 0xFFFFFFFFFFFFFFFF,
                        "control_after_generate": True,
                        "tooltip": "노이즈 생성에 사용되는 랜덤 시드.",
                    },
                ),
            },
            "hidden": {
                "auth_token": "AUTH_TOKEN_COMFY_ORG",
            },
        }

    RETURN_TYPES = ("VIDEO",)
    DESCRIPTION = "Minimax의 API를 사용해 프롬프트를 기반으로 비디오를 생성합니다"
    FUNCTION = "generate_video"
    CATEGORY = "api node/video/Minimax"
    API_NODE = True
    OUTPUT_NODE = True

    def generate_video(
        self,
        prompt_text,
        seed=0,
        model="T2V-01",
        image: torch.Tensor=None, # ImageToVideo에 사용됨
        subject: torch.Tensor=None, # SubjectToVideo에 사용됨
        auth_token=None,
    ):
        '''
        Minimax 노드 간에 사용되는 함수 - 제공된 인자에 따라 T2V, I2V, S2V를 지원합니다.
        '''
        # 이미지가 전달되면 업로드
        image_url = None
        if image is not None:
            image_url = upload_images_to_comfyapi(image, max_images=1, auth_token=auth_token)[0]

        # TODO: 주제 처리 방법을 고민해야 함, API는 S2V-01 모델 사용 시 잘못된 매개변수를 반환함
        subject_reference = None
        if subject is not None:
            subject_url = upload_images_to_comfyapi(subject, max_images=1, auth_token=auth_token)[0]
            subject_reference = [SubjectReferenceItem(image=subject_url)]


        video_generate_operation = SynchronousOperation(
            endpoint=ApiEndpoint(
                path="/proxy/minimax/video_generation",
                method=HttpMethod.POST,
                request_model=MinimaxVideoGenerationRequest,
                response_model=MinimaxVideoGenerationResponse,
            ),
            request=MinimaxVideoGenerationRequest(
                model=Model(model),
                prompt=prompt_text,
                callback_url=None,
                first_frame_image=image_url,
                subject_reference=subject_reference,
                prompt_optimizer=None,
            ),
            auth_token=auth_token,
        )
        response = video_generate_operation.execute()

        task_id = response.task_id
        if not task_id:
            raise Exception(f"Minimax 생성 실패: {response.base_resp}")

        video_generate_operation = PollingOperation(
            poll_endpoint=ApiEndpoint(
                path="/proxy/minimax/query/video_generation",
                method=HttpMethod.GET,
                request_model=EmptyRequest,
                response_model=MinimaxTaskResultResponse,
                query_params={"task_id": task_id},
            ),
            completed_statuses=["Success"],
            failed_statuses=["Fail"],
            status_extractor=lambda x: x.status.value,
            auth_token=auth_token,
        )
        task_result = video_generate_operation.execute()

        file_id = task_result.file_id
        if file_id is None:
            raise Exception("요청이 성공하지 않았습니다. 파일 ID가 없습니다.")
        file_retrieve_operation = SynchronousOperation(
            endpoint=ApiEndpoint(
                path="/proxy/minimax/files/retrieve",
                method=HttpMethod.GET,
                request_model=EmptyRequest,
                response_model=MinimaxFileRetrieveResponse,
                query_params={"file_id": int(file_id)},
            ),
            request=EmptyRequest(),
            auth_token=auth_token,
        )
        file_result = file_retrieve_operation.execute()

        file_url = file_result.file.download_url
        if file_url is None:
            raise Exception(
                f"응답에서 비디오를 찾을 수 없습니다. 전체 응답: {file_result.model_dump()}"
            )
        logging.info(f"생성된 비디오 URL: {file_url}")

        video_io = download_url_to_bytesio(file_url)
        if video_io is None:
            error_msg = f"{file_url}에서 비디오 다운로드 실패"
            logging.error(error_msg)
            raise Exception(error_msg)
        return (VideoFromFile(video_io),)
```
